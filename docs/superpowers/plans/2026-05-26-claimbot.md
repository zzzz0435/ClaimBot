# ClaimBot 實作計畫

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立一個 Python Discord Bot，每 6 小時自動從 IsThereAnyDeal API 取得 Steam 限時免費遊戲，推送 Embed 訊息至指定頻道，並避免重複通知。

**Architecture:** 採用 discord.py Cog 模組化架構。`bot.py` 負責初始化；`cogs/free_games.py` 以 `tasks.loop(hours=6)` 驅動排程與發送；`services/itad_client.py` 封裝 ITAD API 呼叫；`storage/seen_games.py` 負責讀寫 `data/seen_games.json` 以防重複通知。

**Tech Stack:** Python 3.11+、discord.py 2.x、aiohttp 3.9+、python-dotenv、pytest、pytest-asyncio 0.23+、aioresponses

```python
import json
import logging
from pathlib import Path

log = logging.getLogger(__name__)


class SeenGames:
    def __init__(self, path: Path):
        self._path = path
        self.seen_ids: set[str] = self._load()

    def _load(self) -> set[str]:
        if not self._path.exists():
            self._write(set())
            return set()
        try:
            data = json.loads(self._path.read_text(encoding="utf-8"))
            return set(data.get("seen_ids", []))
        except (json.JSONDecodeError, KeyError, ValueError):
            log.error("seen_games.json 格式損毀，重置為空檔案")
            self._write(set())
            return set()

    def _write(self, ids: set[str]) -> None:
        self._path.parent.mkdir(parents=True, exist_ok=True)
        self._path.write_text(
            json.dumps({"seen_ids": sorted(ids)}, ensure_ascii=False, indent=2),
            encoding="utf-8",
        )

    def add(self, new_ids: set[str]) -> None:
        self.seen_ids |= new_ids
        self._write(self.seen_ids)

    def contains(self, game_id: str) -> bool:
        return game_id in self.seen_ids
```

```python
import logging
from dataclasses import dataclass

import aiohttp

log = logging.getLogger(__name__)

ITAD_DEALS_URL = "https://api.isthereanydeal.com/v3/deals"


@dataclass
class FreeGame:
    id: str
    title: str
    url: str
    image_url: str
    expires_at: str | None


class ITADClient:
    def __init__(self, api_key: str):
        self._api_key = api_key

    async def get_free_games(self) -> list[FreeGame]:
        params = {
            "shops": "steam",
            "price_max": 0,
            "limit": 20,
            "key": self._api_key,
        }
        try:
            async with aiohttp.ClientSession() as session:
                async with session.get(
                    ITAD_DEALS_URL,
                    params=params,
                    timeout=aiohttp.ClientTimeout(total=10),
                ) as resp:
                    if resp.status != 200:
                        log.warning("ITAD API 回傳 HTTP %s，跳過本次", resp.status)
                        return []
                    data = await resp.json()
        except aiohttp.ClientError as exc:
            log.warning("ITAD API 請求失敗：%s", exc)
            return []

        return [self._parse(item) for item in data.get("list", [])]

    def _parse(self, item: dict) -> FreeGame:
        assets = item.get("assets", {})
        deal = item.get("deal", {})
        return FreeGame(
            id=item["id"],
            title=item["title"],
            url=deal.get("url", ""),
            image_url=assets.get("banner300") or assets.get("banner145") or "",
            expires_at=deal.get("expiry"),
        )
```

```python
import logging
from pathlib import Path

import discord
from discord.ext import commands, tasks

from services.itad_client import FreeGame, ITADClient
from storage.seen_games import SeenGames

log = logging.getLogger(__name__)

SEEN_GAMES_PATH = Path("data/seen_games.json")
EMBED_COLOR = discord.Color(0x2ECC71)


def filter_new_games(games: list[FreeGame], seen_ids: set[str]) -> list[FreeGame]:
    return [g for g in games if g.id not in seen_ids]


def build_embed(game: FreeGame) -> discord.Embed:
    embed = discord.Embed(title=game.title, url=game.url, color=EMBED_COLOR)
    if game.image_url:
        embed.set_image(url=game.image_url)
    if game.expires_at:
        embed.add_field(name="到期時間", value=game.expires_at, inline=False)
    embed.set_footer(text="資料來源：IsThereAnyDeal")
    return embed


class FreeGamesCog(commands.Cog):
    def __init__(self, bot: commands.Bot, channel_id: int, itad_api_key: str):
        self._bot = bot
        self._channel_id = channel_id
        self._client = ITADClient(api_key=itad_api_key)
        self._seen = SeenGames(SEEN_GAMES_PATH)
        self.check_free_games.start()

    def cog_unload(self):
        self.check_free_games.cancel()

    @tasks.loop(hours=6)
    async def check_free_games(self):
        channel = self._bot.get_channel(self._channel_id)
        if channel is None:
            log.error("找不到頻道 ID %s", self._channel_id)
            return

        games = await self._client.get_free_games()
        new_games = filter_new_games(games, self._seen.seen_ids)

        if not new_games:
            log.info("無新免費遊戲，跳過")
            return

        for game in new_games:
            await channel.send(embed=build_embed(game))

        self._seen.add({g.id for g in new_games})
        log.info("已發送 %d 款新免費遊戲", len(new_games))

    @check_free_games.before_loop
    async def before_check(self):
        await self._bot.wait_until_ready()
```

```python
import logging
import os

import discord
from discord.ext import commands
from dotenv import load_dotenv

from cogs.free_games import FreeGamesCog

load_dotenv()

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
)
log = logging.getLogger(__name__)

TOKEN = os.environ["DISCORD_TOKEN"]
CHANNEL_ID = int(os.environ["DISCORD_CHANNEL_ID"])
ITAD_API_KEY = os.environ["ITAD_API_KEY"]

intents = discord.Intents.default()
bot = commands.Bot(command_prefix="!", intents=intents)


@bot.event
async def on_ready():
    channel = bot.get_channel(CHANNEL_ID)
    if channel is None:
        log.error("無效的 DISCORD_CHANNEL_ID：%s，Bot 停止", CHANNEL_ID)
        await bot.close()
        return
    log.info("Bot 已上線：%s，目標頻道：#%s", bot.user, channel.name)
    await bot.add_cog(FreeGamesCog(bot, CHANNEL_ID, ITAD_API_KEY))


bot.run(TOKEN)
```
